#package 'example1'
#ops 'nolib','mainClass','mainFunc' 
import 'Mod';
const b as Basic; // const不提升
import 'Basic'; // import语句和变量声明会提升
stay b<int> num1; // stay关键字阻止提升
hoist Basic<string> str1; // hoist关键字可以提升语句，提升后排序为语句的出现顺序
class mainClass{
    function<Basic<void>> mainFunc(){}
}
class apple:Mod<item>{ // 声明apple类并继承Mod<item>类
    // 此处省略内容
}
Registry.egistered(&apple); // 注册apple类并传入类的对象化
function<Basic<void>> nothing(){
    b<int> num666;
}
// 此处无法使用num666因为它属于函数内部。
function<b<int>> aNumberFunction(b<int> num1,b<int> num2){ /* 作用域的容器重名并耦合时，优先级如下
上级作用域<下级作用域
函数外容器<函数内部容器<函数形参
任何作用域的容器不能重复声明
    #destoryArea
    // 摧毁作用域，这会导致num3的作用域在上一层作用域————全局作用域
    Basic<int> num3=args.num1+args.num2; // args是实参对象
    return num3;
}
// 此处num3声明但未赋值，形成暂时性死区，不能使用
aNumberFunction(1,2); // 3
// 此处num3的值为3
